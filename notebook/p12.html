y_train_array = np.array(y_train).astype(int)

pipeline = make_pipeline(StandardScaler(), KMeans(n_clusters=2))

clusters = pipeline.fit_predict(X_train)

def align_clusters_to_labels(clusters, true_labels):
    mapping = {}
    for cluster in np.unique(clusters):
        mask = clusters == cluster
        label_values = true_labels[mask]
        most_common = np.bincount(label_values).argmax()
        mapping[cluster] = most_common
    return np.vectorize(mapping.get)(clusters)

clusters_aligned = align_clusters_to_labels(clusters, y_train_array)

accuracy = accuracy_score(y_train_array, clusters_aligned)
recall = recall_score(y_train_array, clusters_aligned)
auc = roc_auc_score(y_train_array, clusters_aligned)

print("Précision :", accuracy)
print("Rappel:", recall)
print("AUC:", auc)
ConfusionMatrixDisplay.from_predictions(y_train_array, clusters_aligned)
